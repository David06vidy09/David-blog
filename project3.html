Project 1: Separate Even and Odd Numbers

Code

#include <stdio.h>

int main() {
    int array[100], evenArray[100], oddArray[100];
    int n, i, evenCount = 0, oddCount = 0;

    // Validate number of elements
    do {
        printf("Input the number of elements to be stored in the array (1-100): ");
        scanf("%d", &n);
        if (n < 1 || n > 100) {
            printf("Invalid input! Please enter a number between 1 and 100.\n");
        }
    } while (n < 1 || n > 100);

    // Input elements (index starting from 1)
    for (i = 1; i <= n; i++) {
        printf("element - %d : ", i);
        scanf("%d", &array[i]);
    }

    // Separate into even and odd arrays
    for (i = 1; i <= n; i++) {
        if (array[i] % 2 == 0) {
            evenArray[evenCount] = array[i];
            evenCount++;
        } else {
            oddArray[oddCount] = array[i];
            oddCount++;
        }
    }

    // Print Even Array
    printf("The elements of EvenArray are : ");
    for (i = 0; i < evenCount; i++) {
        printf("%d ", evenArray[i]);
    }
    printf("\n");

    // Print Odd Array
    printf("The elements of OddArray are : ");
    for (i = 0; i < oddCount; i++) {
        printf("%d ", oddArray[i]);
    }
    printf("\n");

    return 0;
}

Project 2: Find Maximum and Minimum

Code

#include <stdio.h>

int main() {
    int array[100], n, i;
    int max, min;

    // Validate number of elements
    do {
        printf("Input the number of elements to be stored in the array (1-100): ");
        scanf("%d", &n);
        if (n < 1 || n > 100) {
            printf("Invalid input! Please enter a number between 1 and 100.\n");
        }
    } while (n < 1 || n > 100);

    // Input elements (index starting from 0)
    for (i = 0; i < n; i++) {
        printf("element - %d : ", i + 1);
        scanf("%d", &array[i]);
    }

    // Initialize max and min
    max = array[0];
    min = array[0];

    // Find max and min
    for (i = 1; i < n; i++) {
        if (array[i] > max)
            max = array[i];
        if (array[i] < min)
            min = array[i];
    }

    // Output results
    printf("Maximum element is : %d\n", max);
    printf("Minimum element is : %d\n", min);

    return 0;
}

Project 3: Grade Evaluation

Code

// Members: Avenido, Ciroy, Diabil, Piloton
#include <stdio.h>

int main() {
    int grade;
    int result;

    do {
        printf("Enter a grade (1-100) or 0 to exit: ");
        result = scanf("%d", &grade);

        if (result != 1) {
            // Clear invalid input
            while (getchar() != '\n');
            printf("Please only input integers from 1-100.\n");
            continue;
        }

        if (grade == 0) {
            printf("Exiting...\n");
            break;
        }

        if (grade < 1 || grade > 100) {
            printf("Invalid input. Please enter a number between 1 and 100.\n");
            continue;
        }

        // Grade evaluation
        if (grade >= 95) {
            printf("With High Honors\n");
        } else if (grade >= 90) {
            printf("With Honors\n");
        } else if (grade >= 75) {
            printf("You Passed\n");
        } else {
            printf("You Failed\n");
        }

    } while (1);

    return 0;
}
